  6   # A Dataset represents a view of a the data in a database, constrained by
  7   # specific parameters such as filtering conditions, order, etc. Datasets
  8   # can be used to create, retrieve, update and delete records.
  9   # 
 10   # Query results are always retrieved on demand, so a dataset can be kept
 11   # around and reused indefinitely:
 12   #
 13   #   my_posts = DB[:posts].filter(:author => 'david') # no records are retrieved
 14   #   p my_posts.all # records are now retrieved
 15   #   ...
 16   #   p my_posts.all # records are retrieved again
 17   #
 18   # In order to provide this functionality, dataset methods such as where, 
 19   # select, order, etc. return modified copies of the dataset, so you can
 20   # use different datasets to access data:
 21   #
 22   #   posts = DB[:posts]
 23   #   davids_posts = posts.filter(:author => 'david')
 24   #   old_posts = posts.filter('stamp < ?', Date.today - 7)
 25   #
 26   # Datasets are Enumerable objects, so they can be manipulated using any
 27   # of the Enumerable methods, such as map, inject, etc.
 28   #
 29   # === Methods added via metaprogramming

 31   # Some methods are added via metaprogramming:
 32   #
 33   # * ! methods - These methods are the same as their non-! counterparts,
 34   #   but they modify the receiver instead of returning a modified copy

273     # nil is specified, the dataset is turned into a naked dataset and returns
274     # records as hashes. If a model class specified, the dataset is modified
275     # to return records as instances of the model class, e.g:
276     #
277     #   class MyModel
278     #     def initialize(values)
279     #       @values = values
280     #       ...
281     #     end
282     #   end
283     # 
284     #   dataset.set_model(MyModel)
285     #
286     # You can also provide additional arguments to be passed to the model's
287     # initialize method:
288     #
289     #   class MyModel
290     #     def initialize(values, options)
291     #       @values = values
292     #       ...
293     #     end
294     #   end
295     # 
296     #   dataset.set_model(MyModel, :allow_delete => false)
297     #  
298     # The dataset can be made polymorphic by specifying a column name as the
299     # polymorphic key and a hash mapping column values to model classes.
300     #
301     #   dataset.set_model(:kind, {1 => Person, 2 => Business})
302     #
303     # You can also set a default model class to fall back on by specifying a
304     # class corresponding to nil:
305     #
306     #   dataset.set_model(:kind, {nil => DefaultClass, 1 => Person, 2 => Business})
307     #
308     # To make sure that there is always a default model class, the hash provided
309     # should have a default value.  To make the dataset map string values to
310     # model classes, and keep a good default, try:
311     #
312     #   dataset.set_model(:kind, Hash.new{|h,k| h[k] = (k.constantize rescue DefaultClass)})
313     def set_model(key, *args)
314       # This code is more verbose then necessary for performance reasons
315       case key
316       when nil # set_model(nil) => no argument provided, so the dataset is denuded
317         @opts.merge!(:naked => true, :models => nil, :polymorphic_key => nil)
318         self.row_proc = nil
319       when Class
320         # isomorphic model
321         @opts.merge!(:naked => nil, :models => {nil => key}, :polymorphic_key => nil)
322         if key.respond_to?(:load)
323           # the class has a values setter method, so we use it
324           self.row_proc = proc{|h| key.load(h, *args)}
325         else
326           # otherwise we just pass the hash to the constructor
327           self.row_proc = proc{|h| key.new(h, *args)}
328         end
329       when Symbol
330         # polymorphic model
331         hash = args.shift || raise(ArgumentError, "No class hash supplied for polymorphic model")
332         @opts.merge!(:naked => true, :models => hash, :polymorphic_key => key)
333         if (hash.empty? ? (hash[nil] rescue nil) : hash.values.first).respond_to?(:load)
334           # the class has a values setter method, so we use it
335           self.row_proc = proc do |h|
336             c = hash[h[key]] || hash[nil] || \
337               raise(Error, "No matching model class for record (#{polymorphic_key} => #{h[polymorphic_key].inspect})")
338             c.load(h, *args)
339           end
340         else
341           # otherwise we just pass the hash to the constructor
342           self.row_proc = proc do |h|
343             c = hash[h[key]] || hash[nil] || \
344               raise(Error, "No matching model class for record (#{polymorphic_key} => #{h[polymorphic_key].inspect})")
345             c.new(h, *args)
346           end
347         end
348       else
349         raise ArgumentError, "Invalid model specified"
350       end
351       self
352     end

360     # Sets a value transform which is used to convert values loaded and saved
361     # to/from the database. The transform should be supplied as a hash. Each
362     # value in the hash should be an array containing two proc objects - one
363     # for transforming loaded values, and one for transforming saved values.
364     # The following example demonstrates how to store Ruby objects in a dataset
365     # using Marshal serialization:
366     #
367     #   dataset.transform(:obj => [
368     #     proc {|v| Marshal.load(v)},
369     #     proc {|v| Marshal.dump(v)}
370     #   ])
371     #
372     #   dataset.insert_sql(:obj => 1234) #=>
373     #   "INSERT INTO items (obj) VALUES ('\004\bi\002\322\004')"
374     #
375     # Another form of using transform is by specifying stock transforms:
376     # 
377     #   dataset.transform(:obj => :marshal)
378     #
379     # The currently supported stock transforms are :marshal and :yaml.
380     def transform(t)
381       @transform = t
382       t.each do |k, v|
383         case v
384         when Array
385           if (v.size != 2) || !v.first.is_a?(Proc) && !v.last.is_a?(Proc)
386             raise Error::InvalidTransform, "Invalid transform specified"
387           end
388         else
389           unless v = STOCK_TRANSFORMS[v]
390             raise Error::InvalidTransform, "Invalid transform specified"
391           else
392             t[k] = v
393           end
394         end
395       end
396       self
397     end

399     # Applies the value transform for data loaded from the database.
400     def transform_load(r)
401       r.inject({}) do |m, kv|
402         k, v = *kv
403         m[k] = (tt = @transform[k]) ? tt[0][v] : v
