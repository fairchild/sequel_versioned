123     # Returns a copy of the dataset with the given conditions imposed upon it.  
124     # If the query has been grouped, then the conditions are imposed in the 
125     # HAVING clause. If not, then they are imposed in the WHERE clause. Filter
126     # 
127     # filter accepts the following argument types:
128     #
129     # * Hash - list of equality expressions
130     # * Array - depends:
131     #   * If first member is a string, assumes the rest of the arguments
132     #     are parameters and interpolates them into the string.
133     #   * If all members are arrays of length two, treats the same way
134     #     as a hash, except it allows for duplicate keys to be
135     #     specified.
136     # * String - taken literally
137     # * Symbol - taken as a boolean column argument (e.g. WHERE active)
138     # * Sequel::SQL::BooleanExpression - an existing condition expression,
139     #   probably created using the Sequel blockless filter DSL.
140     #
141     # filter also takes a block, which should return one of the above argument
142     # types, and is treated the same way. If both a block and regular argument
143     # are provided, they get ANDed together.
144     #
145     # Examples:
146     #
147     #   dataset.filter(:id => 3).sql #=>
148     #     "SELECT * FROM items WHERE (id = 3)"
149     #   dataset.filter('price < ?', 100).sql #=>
150     #     "SELECT * FROM items WHERE price < 100"
151     #   dataset.filter([[:id, (1,2,3)], [:id, 0..10]]).sql #=>
152     #     "SELECT * FROM items WHERE ((id IN (1, 2, 3)) AND ((id >= 0) AND (id <= 10)))"
153     #   dataset.filter('price < 100').sql #=>
154     #     "SELECT * FROM items WHERE price < 100"
155     #   dataset.filter(:active).sql #=>
156     #     "SELECT * FROM items WHERE :active
157     #   dataset.filter(:price < 100).sql #=>
158     #     "SELECT * FROM items WHERE (price < 100)"
159     # 
160     # Multiple filter calls can be chained for scoping:
161     #
162     #   software = dataset.filter(:category => 'software')
163     #   software.filter(price < 100).sql #=>
164     #     "SELECT * FROM items WHERE ((category = 'software') AND (price < 100))"
165     #
166     # See doc/dataset_filters.rdoc for more examples and details.
167     def filter(*cond, &block)
168       clause = (@opts[:having] ? :having : :where)
169       cond = cond.first if cond.size == 1
170       cond = transform_save(cond) if @transform if cond.is_a?(Hash)
171       cond = filter_expr(cond, &block)
172       cond = SQL::BooleanExpression.new(:AND, @opts[clause], cond) if @opts[clause] && !@opts[clause].blank?

174     end

353     #
354     # * type - The type of join to do (:inner, :left_outer, :right_outer, :full)
355     # * table - Depends on type:
356     #   * Dataset - a subselect is performed with an alias of tN for some value of N
357     #   * Model (or anything responding to :table_name) - table.table_name
358     #   * String, Symbol: table
359     # * expr - specifies conditions, depends on type:
360     #   * Hash, Array with all two pairs - Assumes key (1st arg) is column of joined table (unless already
361     #     qualified), and value (2nd arg) is column of the last joined or primary table.
362     #     To specify multiple conditions on a single joined table column, you must use an array.
363     #     Uses a JOIN with an ON clause.
364     #   * Array - If all members of the array are symbols, considers them as columns and 
365     #     uses a JOIN with a USING clause.  Most databases will remove duplicate columns from
366     #     the result set if this is used.
367     #   * nil - If a block is not given, doesn't use ON or USING, so the JOIN should be a NATURAL
368     #     or CROSS join. If a block is given, uses a ON clause based on the block, see below.
369     #   * Everything else - pretty much the same as a using the argument in a call to filter,
370     #     so strings are considered literal, symbols specify boolean columns, and blockless
371     #     filter expressions can be used. Uses a JOIN with an ON clause.
372     # * table_alias - the name of the table's alias when joining, necessary for joining
373     #   to the same table more than once.  No alias is used by default.
374     # * block - The block argument should only be given if a JOIN with an ON clause is used,
375     #   in which case it yields the table alias/name for the table currently being joined,
376     #   the table alias/name for the last joined (or first table), and an array of previous
377     #   SQL::JoinClause.
378     def join_table(type, table, expr=nil, table_alias=nil, &block)
379       if Dataset === table
380         if table_alias.nil?
381           table_alias_num = (@opts[:num_dataset_sources] || 0) + 1
382           table_alias = "t#{table_alias_num}"
383         end
384         table_name = table_alias
385       else
386         table = table.table_name if table.respond_to?(:table_name)
387         table_name = table_alias || table
388       end
389 
390       join = if expr.nil? and !block_given?
391         SQL::JoinClause.new(type, table, table_alias)
392       elsif Array === expr and !expr.empty? and expr.all?{|x| Symbol === x}
393         raise(Sequel::Error, "can't use a block if providing an array of symbols as expr") if block_given?
394         SQL::JoinUsingClause.new(expr, type, table, table_alias)
395       else
396         last_alias = @opts[:last_joined_table] || first_source
397         if Hash === expr or (Array === expr and expr.all_two_pairs?)
398           expr = expr.collect do |k, v|
399             k = qualified_column_name(k, table_name) if k.is_a?(Symbol)
400             v = qualified_column_name(v, last_alias) if v.is_a?(Symbol)

403         end
404         if block_given?
405           expr2 = yield(table_name, last_alias, @opts[:join] || [])
406           expr = expr ? SQL::BooleanExpression.new(:AND, expr, expr2) : expr2
407         end
408         SQL::JoinOnClause.new(expr, type, table, table_alias)
409       end

412       opts[:num_dataset_sources] = table_alias_num if table_alias_num

777     # SQL fragment based on the expr type.  See #filter.
778     def filter_expr(expr = nil, &block)
779       expr = nil if expr == []
780       if expr && block
781         return SQL::BooleanExpression.new(:AND, filter_expr(expr), filter_expr(block))
782       elsif block
783         expr = block
784       end
785       case expr
786       when Hash
787         SQL::BooleanExpression.from_value_pairs(expr)
788       when Array
789         if String === expr[0]
790           filter_expr(expr.shift.gsub(QUESTION_MARK){literal(expr.shift)}.lit)
791         else
792           SQL::BooleanExpression.from_value_pairs(expr)
793         end
794       when Proc
795         filter_expr(expr.call(SQL::VirtualRow.new))
796       when SQL::NumericExpression, SQL::StringExpression
797         raise(Error, "Invalid SQL Expression type: #{expr.inspect}") 
798       when Symbol, SQL::Expression
799         expr
800       when TrueClass, FalseClass
801         SQL::BooleanExpression.new(:NOOP, expr)
802       when String
803         "(#{expr})".lit
804       else
805         raise(Error, 'Invalid filter argument')
806       end
807     end

870     # Splits the symbol into three parts.  Each part will
871     # either be a string or nil.
872     #
873     # For columns, these parts are the table, column, and alias.
874     # For tables, these parts are the schema, table, and alias.
875     def split_symbol(sym)
876       s = sym.to_s
877       if m = COLUMN_REF_RE1.match(s)
878         m[1..3]
879       elsif m = COLUMN_REF_RE2.match(s)
880         [nil, m[1], m[2]]
881       elsif m = COLUMN_REF_RE3.match(s)
882         [m[1], m[2], nil]
