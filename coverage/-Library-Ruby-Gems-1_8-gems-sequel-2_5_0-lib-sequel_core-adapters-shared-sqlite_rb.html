 86       # SQLite supports schema parsing using the table_info PRAGMA, so
 87       # parse the output of that into the format Sequel expects.
 88       def schema_parse_table(table_name, opts)
 89         rows = self["PRAGMA table_info(?)", table_name].collect do |row|
 90           row.delete(:cid)
 91           row[:column] = row.delete(:name)
 92           row[:allow_null] = row.delete(:notnull).to_i == 0 ? 'YES' : 'NO'
 93           row[:default] = row.delete(:dflt_value)
 94           row[:primary_key] = row.delete(:pk).to_i == 1 ? true : false 
 95           row[:db_type] = row.delete(:type)
 96           if m = SCHEMA_TYPE_RE.match(row[:db_type])
 97             row[:db_type] = m[1]
 98             row[:max_chars] = m[2].to_i
 99           else
100              row[:max_chars] = nil
101           end
102           row[:numeric_precision] = nil
103           row
104         end
105         schema_parse_rows(rows)
106       end

119       # SQLite does not support pattern matching via regular expressions.
120       # SQLite is case insensitive (depending on pragma), so use LIKE for
121       # ILIKE.
122       def complex_expression_sql(op, args)
123         case op
124         when :~, :'!~', :'~*', :'!~*'
125           raise Error, "SQLite does not support pattern matching via regular expressions"
126         when :LIKE, :'NOT LIKE', :ILIKE, :'NOT ILIKE'

128           "#{'NOT ' if [:'NOT LIKE', :'NOT ILIKE'].include?(op)}(#{literal(args.at(0))} LIKE #{literal(args.at(1))})"
129         else
130           super(op, args)
131         end
