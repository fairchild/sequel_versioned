455     def run_association_callbacks(reflection, callback_type, object)
456       raise_error = @raise_on_save_failure

499     # Returns all methods that can be used for attribute
500     # assignment (those that end with =), modified by the only
501     # and except arguments:
502     #
503     # * only
504     #   * false - Don't modify the results
505     #   * nil - if the model has allowed_columns, use only these, otherwise, don't modify
506     #   * Array - allow only the given methods to be used
507     # * except
508     #   * false - Don't modify the results
509     #   * nil - if the model has restricted_columns, remove these, otherwise, don't modify
510     #   * Array - remove the given methods
511     #
512     # only takes precedence over except, and if only is not used, certain methods are always
513     # restricted (RESTRICTED_SETTER_METHODS).  The primary key is restricted by default as
514     # well, see Model.unrestrict_primary_key to change this.
515     def setter_methods(only, except)
516       only = only.nil? ? model.allowed_columns : only
517       except = except.nil? ? model.restricted_columns : except
518       if only
519         only.map{|x| "#{x}="}
520       else
521         meths = methods.collect{|x| x.to_s}.grep(/=\z/) - RESTRICTED_SETTER_METHODS
522         meths -= Array(primary_key).map{|x| "#{x}="} if primary_key && model.restrict_primary_key?
523         meths -= except.map{|x| "#{x}="} if except
524         meths
525       end
526     end
