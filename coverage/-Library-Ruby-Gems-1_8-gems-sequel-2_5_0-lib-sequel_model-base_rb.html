448     # Get the schema from the database, fall back on checking the columns
449     # via the database if that will return inaccurate results or if
450     # it raises an error.
451     def self.get_db_schema(reload = false) # :nodoc:
452       set_columns(nil)
453       return nil unless @dataset
454       schema_hash = {}
455       ds_opts = dataset.opts
456       single_table = ds_opts[:from] && (ds_opts[:from].length == 1) \
457         && !ds_opts.include?(:join) && !ds_opts.include?(:sql)
458       get_columns = proc{columns rescue []}
459       if single_table && (schema_array = (db.schema(table_name, :reload=>reload) rescue nil))
460         schema_array.each{|k,v| schema_hash[k] = v}
461         if ds_opts.include?(:select)
462           # Dataset only selects certain columns, delete the other
463           # columns from the schema
464           cols = get_columns.call
465           schema_hash.delete_if{|k,v| !cols.include?(k)}
466           cols.each{|c| schema_hash[c] ||= {}}
467         else
468           # Dataset is for a single table with all columns,
469           # so set the columns based on the order they were
470           # returned by the schema.
471           cols = schema_array.collect{|k,v| k}
472           set_columns(cols)
473           # Also set the columns for the dataset, so the dataset
474           # doesn't have to do a query to get them.
475           dataset.instance_variable_set(:@columns, cols)
476         end
477       else
478         # If the dataset uses multiple tables or custom sql or getting
479         # the schema raised an error, just get the columns and
480         # create an empty schema hash for it.
481         get_columns.call.each{|c| schema_hash[c] = {}}
482       end
483       schema_hash
484     end
