307     # unique in the database.  You should also add a unique index in the
308     # database, as this suffers from a fairly obvious race condition.
309     #
310     # Possible Options:
311     # * :allow_nil - Whether to skip the validation if the value(s) is/are nil (default: false)
312     # * :if - A symbol (indicating an instance_method) or proc (which is instance_evaled)
313     #   skipping this validation if it returns nil or false.
314     # * :message - The message to use (default: 'is already taken')
315     # * :tag - The tag to use for this validation (default: :uniqueness)
316     def self.validates_uniqueness_of(*atts)
317       opts = {
318         :message => 'is already taken',
319       }.merge!(atts.extract_options!)
320 
321       atts << {:if=>opts[:if], :tag=>opts[:tag]||:uniqueness}
322       validates_each(*atts) do |o, a, v|
323         error_field = a
324         a = Array(a)
325         v = Array(v)
326         next unless v.any? or opts[:allow_nil] == false
327         ds = o.class.filter(a.zip(v))
328         num_dups = ds.count
329         allow = if num_dups == 0
330           # No unique value in the database
331           true
332         elsif num_dups > 1
333           # Multiple "unique" values in the database!!
334           # Someone didn't add a unique index
335           false
336         elsif o.new?
337           # New record, but unique value already exists in the database
338           false
339         elsif ds.first === o
340           # Unique value exists in database, but for the same record, so the update won't cause a duplicate record
341           true
342         else
343           false
344         end
345         o.errors[error_field] << opts[:message] unless allow
346       end
