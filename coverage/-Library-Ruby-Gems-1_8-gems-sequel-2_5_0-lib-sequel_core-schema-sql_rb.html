 63       def column_definition_sql(column)
 64         return constraint_definition_sql(column) if column[:type] == :check
 65         sql = "#{quote_identifier(column[:name])} #{type_literal(column)}"
 66         sql << UNIQUE if column[:unique]

316           (schemas[row.delete(:table_name).to_sym] ||= []) << row

327         schema = []

330           row[:default] = nil if row[:default].blank?
