359     # A simple implementation of SQL transactions. Nested transactions are not 
360     # supported - calling #transaction within a transaction will reuse the 
361     # current transaction. Should be overridden for databases that support nested 
362     # transactions.
363     def transaction(server=nil)
364       synchronize(server) do |conn|
365         return yield(conn) if @transactions.include?(Thread.current)
366         log_info(SQL_BEGIN)
367         conn.execute(SQL_BEGIN)
368         begin
369           @transactions << Thread.current
370           yield(conn)
371         rescue Exception => e
372           log_info(SQL_ROLLBACK)
373           conn.execute(SQL_ROLLBACK)
374           transaction_error(e)
375         ensure
376           unless e
377             log_info(SQL_COMMIT)
378             conn.execute(SQL_COMMIT)
379           end
380           @transactions.delete(Thread.current)
381         end
382       end
383     end

385     # Typecast the value to the given column_type. Can be overridden in
386     # adapters to support database specific column types.
387     def typecast_value(column_type, value)
388       return nil if value.nil?
389       case column_type
390       when :integer
391         Integer(value)
392       when :string
393         value.to_s
394       when :float
395         Float(value)
396       when :decimal
397         case value
398         when BigDecimal
399           value
400         when String, Float
401           value.to_d
402         when Integer
403           value.to_s.to_d
404         else
405           raise ArgumentError, "invalid value for BigDecimal: #{value.inspect}"
406         end
407       when :boolean
408         case value
409         when false, 0, "0", /\Af(alse)?\z/i
410           false
411         else
412           value.blank? ? nil : true
413         end
414       when :date
415         case value
416         when Date
417           value
418         when DateTime, Time
419           Date.new(value.year, value.month, value.day)
420         when String
421           value.to_date
422         else
423           raise ArgumentError, "invalid value for Date: #{value.inspect}"
424         end
425       when :time
426         case value
427         when Time
428           value
429         when String
430           value.to_time
431         else
432           raise ArgumentError, "invalid value for Time: #{value.inspect}"
433         end
434       when :datetime
435         raise(ArgumentError, "invalid value for #{tc}: #{value.inspect}") unless value.is_one_of?(DateTime, Date, Time, String)
436         if Sequel.datetime_class === value
437           # Already the correct class, no need to convert
438           value
439         else
440           # First convert it to standard ISO 8601 time, then
441           # parse that string using the time class.
442           (Time === value ? value.iso8601 : value.to_s).to_sequel_time
443         end
444       when :blob
445         value.to_blob
446       else
447         value

477     def server_opts(server)
478       opts = if @opts[:servers] && server_options = @opts[:servers][server]
