 77   # Chooses the first available connection to the given server, or if none are
 78   # available, creates a new connection.  Passes the connection to the supplied
 79   # block:
 80   # 
 81   #   pool.hold {|conn| conn.execute('DROP TABLE posts')}
 82   # 
 83   # Pool#hold is re-entrant, meaning it can be called recursively in
 84   # the same thread without blocking.
 85   #
 86   # If no connection is immediately available and the pool is already using the maximum
 87   # number of connections, Pool#hold will block until a connection
 88   # is available or the timeout expires.  If the timeout expires before a
 89   # connection can be acquired, a Sequel::Error::PoolTimeoutError is 
 90   # raised.
 91   def hold(server=:default)
 92     begin
 93       t = Thread.current
 94       time = Time.new
 95       timeout = time + @timeout
 96       sleep_time = @sleep_time
 97       if conn = owned_connection(t, server)
 98         return yield(conn)
 99       end
100       until conn = acquire(t, server)
101         raise(::Sequel::Error::PoolTimeoutError) if Time.new > timeout
102         sleep sleep_time
103       end
104       begin
105         yield conn
106       ensure
107         release(t, conn, server)
108       end
109     rescue Exception => e
110       raise(@convert_exceptions && !e.is_a?(StandardError) ? RuntimeError.new(e.message) : e)
111     end
112   end
